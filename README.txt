javac -classpath ../opencsv-5.7.1.jar:. inalco/projet_application/*.java
- EXECUTER LA LIGNE DE COMMANDE SUIVANTE POUR LANCER LE PROGRAMME :
java -classpath ../opencsv-5.7.1.jar:. inalco.projet_application.Application
